cmake_minimum_required(VERSION 3.12)

# vgm2wav_adpcm_only executable
add_executable(vgm2wav_adpcm_only
    src/vgm2wav_adpcm_only.cpp
)

# Include directories
target_include_directories(vgm2wav_adpcm_only PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/libvgm
    D:/working/vscode-projects/modizer-libs/libvgm-modizer/libvgm
)

# Link with libvgm libraries
target_link_libraries(vgm2wav_adpcm_only
    ${CMAKE_CURRENT_BINARY_DIR}/lib/libvgm-player.a
    ${CMAKE_CURRENT_BINARY_DIR}/lib/libvgm-emu.a
    ${CMAKE_CURRENT_BINARY_DIR}/lib/libvgm-utils.a
    z
)

# C++ standard
set_target_properties(vgm2wav_adpcm_only PROPERTIES
    CXX_STANDARD 11
    CXX_STANDARD_REQUIRED ON
)
